public OnPlayerClickPlayerTextDraw(playerid, PlayerText:playertextid)
{
    if(playertextid == bank_bg[5][playerid]) return IfPlayerClickDeposit(playerid);
    if(playertextid == bank_deposit[6][playerid]) return OnPlayerClickDepositText(playerid);
    if(playertextid == bank_deposit[5][playerid]) return IfPlayerDeposit(playerid);
    if(playertextid == bank_bg[6][playerid]) return IfPlayerClickSaq(playerid);
    if(playertextid == bank_saq[6][playerid]) return OnPlayerClickSaqText(playerid);
    if(playertextid == bank_saq[5][playerid]) return IfPlayerSaq(playerid);
	if(playertextid == bank_bg[7][playerid]) return IfPlayerClickTransfer(playerid);
	if(playertextid == bank_transfer[10][playerid]) return OnPlayerClickTransferIdText(playerid);
	if(playertextid == bank_transfer[6][playerid]) return OnPlayerClickTransferText(playerid);
	if(playertextid == bank_transfer[5][playerid]) return IfPlayerTransfer(playerid);
	if(playertextid == bank_bg[11][playerid]) return OnPlayerCloseBank(playerid);
	if(playertextid == bank_rl[9][playerid]) return IfPlayerClickLogin(playerid);
	if(playertextid == bank_rl[10][playerid]) return IfPlayerClickRegister(playerid);
	if(playertextid == bank_rl[11][playerid])
	{
		for(new i = 0; i < sizeof(bank_rl); i++) { PlayerTextDrawHide(playerid, bank_rl[i][playerid]); }
		CancelSelectTextDraw(playerid);
	}
    return 1;
}

IfPlayerDeposit(playerid)
{
	if(GetPVarInt(playerid, "quantidade") == 0)
	return SendClientMessage(playerid, -1, ""c_firebrick"Você não depositou nada.");
	GivePlayerMoney(playerid, - GetPVarInt(playerid, "quantidade"));
	p_saldo[playerid] += GetPVarInt(playerid, "quantidade");
	SendClientMessageEx(playerid, -1, ""c_dimgrey"Você depositou "c_seagreen"%dR$ "c_dimgrey"em sua conta corrente.", GetPVarInt(playerid, "quantidade"));
	SendClientMessageEx(playerid, -1, ""c_dimgrey"Seu novo saldo é de "c_seagreen"%dR$"c_dimgrey".", p_saldo[playerid]);
	format(quantia_str, sizeof(quantia_str), "VOCE POSSUI ~g~%dR$ ~w~EM SUA CONTA CORRENTE.", p_saldo[playerid]);
	PlayerTextDrawSetString(playerid, bank_bg[10][playerid], quantia_str);
	return SalvarDados(playerid), SetPVarInt(playerid, "quantidade", 0), format(quantia_str, sizeof(quantia_str), "%d", GetPVarInt(playerid, "quantidade")), PlayerTextDrawSetString(playerid, bank_deposit[6][playerid], quantia_str);
}

IfPlayerSaq(playerid)
{
	if(GetPVarInt(playerid, "quantidade") == 0)
	return SendClientMessage(playerid, -1, ""c_firebrick"Você não sacou nada.");
	GivePlayerMoney(playerid, GetPVarInt(playerid, "quantidade"));
	p_saldo[playerid] -= GetPVarInt(playerid, "quantidade");
	SendClientMessageEx(playerid, -1, ""c_dimgrey"Você "c_firebrick"sacou "c_seagreen"%dR$ "c_dimgrey"de sua conta corrente.", GetPVarInt(playerid, "quantidade"));

}

IfPlayerTransfer(playerid)
{
	if(GetPVarInt(playerid, "quantidade") == 0)
	return SendClientMessage(playerid, -1, ""c_firebrick"Você não transferiu nada.");
	p_saldo[playerid] -= GetPVarInt(playerid, "quantidade");
	p_saldo[GetPVarInt(playerid, "id")] += GetPVarInt(playerid, "quantidade");
	SendClientMessageEx(playerid, -1, ""c_dimgrey"Você transferiu "c_seagreen"%dR$ "c_dimgrey"da sua conta corrente para conta do player %s.", GetPVarInt(playerid, "quantidade"), p_name[GetPVarInt(playerid, "id")]);
	SendClientMessageEx(playerid, -1, ""c_dimgrey"Seu novo saldo é de "c_seagreen"%dR$"c_dimgrey".", p_saldo[playerid]);
	SendClientMessageEx(GetPVarInt(playerid, "id"), -1, ""c_dimgrey"Você recebeu uma transferencia bancária de "c_seagreen"%dR$ "c_dimgrey"do player %s.", GetPVarInt(playerid, "quantidade"), p_name[playerid]);
	SendClientMessageEx(playerid, GetPVarInt(playerid, "id"), ""c_dimgrey"Seu novo saldo é de "c_seagreen"%dR$"c_dimgrey".", p_saldo[GetPVarInt(playerid, "id")]);
	format(quantia_str, sizeof(quantia_str), "VOCE POSSUI ~g~%dR$ ~w~EM SUA CONTA CORRENTE.", p_saldo[playerid]);
	PlayerTextDrawSetString(playerid, bank_bg[10][playerid], quantia_str);
	return SalvarDados(playerid), SalvarDados(GetPVarInt(playerid, "id")), SetPVarInt(playerid, "quantidade", 0), SetPVarInt(playerid, "id", -1), format(quantia_str, sizeof(quantia_str), "%d", GetPVarInt(playerid, "quantidade")), PlayerTextDrawSetString(playerid, bank_transfer[6][playerid], quantia_str);
}

public OnDialogResponse(playerid, dialogid, response, listitem, inputtext[])
{
	switch(dialogid)
	{
        case dialog_registro:
        {
            if (!response) return 1;
            if (response)
            {
                format(Arquivo, 50, "Banco/Contas/%s.ini", p_name[playerid]);
                DOF2::CreateFile(Arquivo);
                DOF2::SetString(Arquivo, "Senha", inputtext);
                DOF2::SaveFile();
                new len = strlen(inputtext);
                if (len < 5 || len > 20) return ShowPlayerDialog(playerid, dialog_registro, DIALOG_STYLE_INPUT, "BANCO - REGISTRO", "Senha deve conter no minimo 5 caracteres.", "Entrar", "Sair");
                SendClientMessage(playerid, 0xFF0000A, "Conta criada com sucesso.");
            }
        }
        case dialog_login:
        {
            if (!response) return 1;
            if (response)
            {
                format(Arquivo, 50, "Banco/Contas/%s.ini", p_name[playerid]);
                format(Senha, 20, "%s", strval(inputtext));
                if (strcmp(inputtext, DOF2::GetString(Arquivo, "Senha"), true) == 0)
                {
                    if (!strlen(inputtext))
					return ShowPlayerDialog(playerid, dialog_login, DIALOG_STYLE_INPUT, "BANCO - LOGIN", "Senha errada, digite a senha correta para logar-se.", "Entrar", "Sair"),
                    SendClientMessage(playerid, 0xFF0000AA, "Senha incorreta, digite novamente!!!");
                    SendClientMessage(playerid, 0xFF0000A, "Logado com sucesso.");
					for(new i = 0; i < sizeof(bank_rl); i++) { PlayerTextDrawHide(playerid, bank_rl[i][playerid]); }
					format(quantia_str, sizeof(quantia_str), "SEJA VEM VINDO %s", p_name[playerid]);
					PlayerTextDrawSetString(playerid, bank_bg[10][playerid], quantia_str);
					format(quantia_str, sizeof(quantia_str), "VOCE POSSUI ~g~%dR$ ~w~EM SUA CONTA CORRENTE.", p_saldo[playerid]);
					PlayerTextDrawSetString(playerid, bank_bg[10][playerid], quantia_str);
					for(new i = 0; i < sizeof(bank_bg); i++) { PlayerTextDrawShow(playerid, bank_bg[i][playerid]); }
                }
                else
                {
                    ShowPlayerDialog(playerid, dialog_login, DIALOG_STYLE_INPUT, "BANCO - LOGIN", "Senha errada, insira a senha correta.", "Entrar", "Sair");
                    SendClientMessage(playerid, 0xFF0000AA, "Senha incorreta, digite novamente!!!");
                }
            }
        }
		case dialog_depositar:
		{
		    if(!response) return 1;
			if(GetPlayerMoney(playerid) < strval(inputtext))
			return SendClientMessage(playerid, -1, ""c_firebrick"Você não tem tanto dinheiro assim em mãos."),
			ShowPlayerDialog(playerid, dialog_depositar, DIALOG_STYLE_INPUT, "Banco - DEPOSITAR", "Insira a quantia que você deseja depositar.", "Depositar", "Fechar");
			SetPVarInt(playerid, "quantidade", strval(inputtext));
 			format(quantia_str, sizeof(quantia_str), "%d", GetPVarInt(playerid, "quantidade"));
    		PlayerTextDrawSetString(playerid, bank_deposit[6][playerid], quantia_str);
    
		}
		case dialog_sacar:
		{
		    if(!response) return 1;
			if(p_saldo[playerid] < strval(inputtext))
			return SendClientMessage(playerid, -1, ""c_firebrick"Você não tem tanto dinheiro assim em sua conta corrente."),
			ShowPlayerDialog(playerid, dialog_sacar, DIALOG_STYLE_INPUT, "Banco - SACAR", "Insira a quantia que você deseja sacar.", "Sacar", "Fechar");
			SetPVarInt(playerid, "quantidade", strval(inputtext));
 			format(quantia_str, sizeof(quantia_str), "%d", GetPVarInt(playerid, "quantidade"));
    		PlayerTextDrawSetString(playerid, bank_saq[6][playerid], quantia_str);
		}
		case dialog_transferir_id:
		{
		    if(!response) return 1;
			if(!IsPlayerConnected(strval(inputtext)))
			return SendClientMessage(playerid, -1, ""c_firebrick"Jogador Offline."),
			ShowPlayerDialog(playerid, dialog_transferir_id, DIALOG_STYLE_INPUT, "Banco - TRANSFERENCIA - ID", "Insira o ID do player a receber a transferencia.", "Porximo", "Fechar");
			SetPVarInt(playerid, "id", strval(inputtext));
 			format(quantia_str, sizeof(quantia_str), "%d", GetPVarInt(playerid, "id"));
    		PlayerTextDrawSetString(playerid, bank_transfer[10][playerid], quantia_str);
    		ShowPlayerDialog(playerid, 2943, DIALOG_STYLE_MSGBOX, "AVISO", ""c_orange"Agora ao lado, indique a quantia a ser transferida.", "OK", #);
		}
		case dialog_transferir:
		{
		    if(!response) return 1;
			if(p_saldo[playerid] < strval(inputtext))
			return SendClientMessage(playerid, -1, ""c_firebrick"Você não tem tanto dinheiro assim em sua conta corrente."), SetPVarInt(playerid, "id", -1),
			ShowPlayerDialog(playerid, dialog_transferir, DIALOG_STYLE_INPUT, "Banco - TRANSFERIR - QUANTIA", "Indique abaixo a quantidade a ser transferida.", "Enviar", "Fechar");
			SetPVarInt(playerid, "quantidade", strval(inputtext));
 			format(quantia_str, sizeof(quantia_str), "%d", GetPVarInt(playerid, "quantidade"));
    		PlayerTextDrawSetString(playerid, bank_transfer[6][playerid], quantia_str);
		}
		
	}
	return 1;
}

SalvarDados(playerid)
{
    format(Arquivo, 40, "Banco/Contas/%s.ini", p_name[playerid]);
    if(!DOF2::FileExists(Arquivo)) DOF2::CreateFile(Arquivo);
    DOF2::SetInt(Arquivo, "Saldo", p_saldo[playerid]);
    return DOF2::SaveFile();
}

CarregarDados(playerid)
